    import org.IsmaelSS.model.RoundResult;
    import org.junit.jupiter.api.AfterEach;
    import org.junit.jupiter.api.BeforeEach;
    import org.junit.jupiter.api.Test;

    import java.io.File;
    import java.io.IOException;
    import java.nio.file.Files;
    import java.nio.file.Path;
    import java.nio.file.Paths;
    import java.util.List;

    import static org.junit.jupiter.api.Assertions.*;

    class StatsServiceErrorHandlingTest {

        private static final String STATS_FILE = System.getProperty("user.dir")
                + File.separator + "flashcard-stats.json";

        @BeforeEach
        @AfterEach
        void cleanup() {
            new File(STATS_FILE).delete();
        }

        @Test
        void corruptedFileRecoversGracefully() throws IOException {
            Files.writeString(Paths.get(STATS_FILE), "{{{corrupted garbage data}}}");
            StatsService service = new StatsService();
            assertEquals(0, service.getOverallStats().getTotalAnswered(),
                    "Corrupted file should produce empty stats");
            assertTrue(service.hasError(), "Corrupted file should set error flag");
            assertNotNull(service.getLastError(), "Error message should be set");
        }

        @Test
        void freshStartHasNoError() {
            StatsService service = new StatsService();
            assertFalse(service.hasError(), "Fresh start with no file should have no error");
            assertNull(service.getLastError(), "No error message should be present");
        }

        @Test
        void successfulRoundClearsError() {
            StatsService service = new StatsService();
            service.recordRound(List.of(
                    new RoundResult("matematica", "2+2?", true)
            ));
            assertFalse(service.hasError(),
                    "Successful save should clear error flag");
        }

        @Test
        void errorFlagPersistsUntilCleared() throws IOException {
            // Create corrupted file
            Files.writeString(Paths.get(STATS_FILE), "bad data");
            StatsService service = new StatsService();
            assertTrue(service.hasError());

            // Clear explicitly
            service.clearError();
            assertFalse(service.hasError());
            assertNull(service.getLastError());
        }
    }

| T-05-07 | Elevation of Privilege | ErrorUtil Alert | mitigate | Alert parameters are hardcoded or from internal error messages, not user input |
</threat_model>
